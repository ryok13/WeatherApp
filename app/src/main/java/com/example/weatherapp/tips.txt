package com.example.weatherapp;

import android.content.Intent;
import android.os.Bundle;

import androidx.appcompat.app.AppCompatActivity;
import androidx.recyclerview.widget.LinearLayoutManager;
import androidx.recyclerview.widget.RecyclerView;

import com.example.weatherapp.adapter.LocationAdapter;
import com.example.weatherapp.databinding.ActivityMainBinding;
import com.example.weatherapp.model.LocationItem;

import java.util.ArrayList;
import java.util.List;

public class MainActivity extends AppCompatActivity {
    // ViewBinding
    ActivityMainBinding binding;

    RecyclerView recyclerView;

    // Data Source
    List<LocationItem> locationList;

    LocationAdapter locationAdapter;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        binding = ActivityMainBinding.inflate(getLayoutInflater());
        setContentView(binding.getRoot());

        // Initialize RecyclerView
        recyclerView = binding.recyclerViewLocations;

        // Prepare data source (list of locations)
        locationList = new ArrayList<>();
        locationList.add(new LocationItem("Toronto"));
        locationList.add(new LocationItem("Montreal"));
        locationList.add(new LocationItem("Vancouver"));
        locationList.add(new LocationItem("Banff"));
        locationList.add(new LocationItem("Yellowknife"));
        locationList.add(new LocationItem("Calgary"));
        locationList.add(new LocationItem("Ottawa"));


        // Set LayoutManager
        LinearLayoutManager layoutManager = new LinearLayoutManager(this);
        recyclerView.setLayoutManager(layoutManager);

        // Set Adapter
        locationAdapter = new LocationAdapter(
                getApplicationContext(),
                locationList,
                item -> {
                    // Navigate to WeatherDetailActivity with selected city
                    Intent intent = new Intent(MainActivity.this, WeatherDetailActivity.class);
                    intent.putExtra("city_name", item.getCityName());
                    startActivity(intent);
                }
        );

        recyclerView.setAdapter(locationAdapter);
    }
}
